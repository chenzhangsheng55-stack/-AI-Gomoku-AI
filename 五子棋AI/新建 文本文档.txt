let _ac = null, muted = false;
function AC() {
  if (!_ac) _ac = new (window.AudioContext || window.webkitAudioContext)();
  if (_ac.state === 'suspended') _ac.resume();
  return _ac;
}
function dest(node) { node.connect(AC().destination); return node; }

// Pink-noise buffer source (cached per duration bucket)
const _noiseBufs = {};
function pinkNoiseSrc(dur) {
  const key = dur.toFixed(3);
  if (!_noiseBufs[key]) {
    const ac = AC(), len = Math.ceil(ac.sampleRate * dur);
    const buf = ac.createBuffer(1, len, ac.sampleRate);
    const d = buf.getChannelData(0);
    let b0=0,b1=0,b2=0,b3=0,b4=0,b5=0;
    for (let i=0;i<len;i++){
      const w=Math.random()*2-1;
      b0=.99886*b0+w*.0555179; b1=.99332*b1+w*.0750759;
      b2=.96900*b2+w*.1538520; b3=.86650*b3+w*.3104856;
      b4=.55000*b4+w*.5329522; b5=-.7616*b5-w*.0168980;
      d[i]=(b0+b1+b2+b3+b4+b5+w*.5362)*0.1;
    }
    _noiseBufs[key] = buf;
  }
  const src = AC().createBufferSource();
  src.buffer = _noiseBufs[key];
  return src;
}

// ── Stone clack: woody thud with transient ──
function sfxStone(isBlack) {
  if (muted) return;
  const ac = AC(), t = ac.currentTime + 0.008;

  // 1. Sharp transient crack (very short noise burst)
  {
    const src = pinkNoiseSrc(0.05);
    const hp = ac.createBiquadFilter(); hp.type='highpass'; hp.frequency.value=1800;
    const bp = ac.createBiquadFilter(); bp.type='bandpass'; bp.frequency.value=isBlack?2800:3800; bp.Q.value=1.5;
    const g = ac.createGain();
    g.gain.setValueAtTime(0.0001, t);
    g.gain.linearRampToValueAtTime(isBlack?.55:.42, t+.002);
    g.gain.exponentialRampToValueAtTime(0.0001, t+.04);
    src.connect(hp); hp.connect(bp); bp.connect(g); dest(g);
    src.start(t); src.stop(t+.05);
  }
  // 2. Mid wood body resonance
  {
    const src = pinkNoiseSrc(0.1);
    const bp = ac.createBiquadFilter(); bp.type='bandpass'; bp.frequency.value=isBlack?320:500; bp.Q.value=2.2;
    const g = ac.createGain();
    g.gain.setValueAtTime(0.0001, t);
    g.gain.linearRampToValueAtTime(isBlack?.7:.55, t+.003);
    g.gain.exponentialRampToValueAtTime(0.0001, t+.085);
    src.connect(bp); bp.connect(g); dest(g);
    src.start(t); src.stop(t+.1);
  }
  // 3. Low pitch thump
  {
    const o = ac.createOscillator(); o.type='sine';
    o.frequency.setValueAtTime(isBlack?95:145, t);
    o.frequency.exponentialRampToValueAtTime(isBlack?42:65, t+.09);
    const g = ac.createGain();
    g.gain.setValueAtTime(0.0001, t);
    g.gain.linearRampToValueAtTime(isBlack?.42:.32, t+.003);
    g.gain.exponentialRampToValueAtTime(0.0001, t+.1);
    o.connect(g); dest(g); o.start(t); o.stop(t+.12);
  }
}

// ── Undo: soft reversed scrape ──

  if (muted) return;

  // Rising noise sweep
  {
    const src = pinkNoiseSrc(0.12);
    const bp = ac.createBiquadFilter(); bp.type='bandpass';
    bp.frequency.setValueAtTime(600, t);
    bp.frequency.linearRampToValueAtTime(2200, t+.12);
    bp.Q.value = 2;
    const g = ac.createGain();
    g.gain.setValueAtTime(0.0001, t);
    g.gain.linearRampToValueAtTime(.22, t+.02);
    g.gain.linearRampToValueAtTime(0.0001, t+.12);
    src.connect(bp); bp.connect(g); dest(g);
    src.start(t); src.stop(t+.13);
  }
  // Rising pitch glide
  {
    const o = ac.createOscillator(); o.type='sine';

}

// ══════════════════════════════════════════════
// UNDO
// ══════════════════════════════════════════════
function undoMove() {
  if(moveHistory.length===0||gameOver) return;
  // Remove last AI move + last player move (2 plies), or 1 if AI hasn't gone
  const removeCount = (moveHistory.length>=2 && moveHistory[moveHistory.length-1].color===aiColor) ? 2 : 1;
  for(let i=0;i<removeCount;i++) {
    const mv=moveHistory.pop();
    showUndoGhost(mv.r, mv.c, mv.color);
    board[mv.r][mv.c]=0;
  }
  gameOver=false; winLine=[];
  document.getElementById('boardFrame').classList.remove('win-flash');
  if(moveHistory.length>0){
    const last=moveHistory[moveHistory.length-1];
    lastMove=[last.r,last.c]; currentTurn=3-last.color;
  } else { lastMove=null; currentTurn=1; }
  sfxUndo(); updateUndoBtn(); drawBoard(); updateStatus();
}

function drawBoard() {
  ctx.clearRect(0,0,CANVAS_SIZE,CANVAS_SIZE);
  // Wood
  const gr=ctx.createLinearGradient(0,0,CANVAS_SIZE,CANVAS_SIZE);
  gr.addColorStop(0,'#d4a030'); gr.addColorStop(.5,'#be8820'); gr.addColorStop(1,'#a86e10');
  ctx.fillStyle=gr; ctx.fillRect(0,0,CANVAS_SIZE,CANVAS_SIZE);
  // Grain
  ctx.strokeStyle='rgba(0,0,0,.042)'; ctx.lineWidth=.8;
  for(let i=-20;i<CANVAS_SIZE+20;i+=9){ctx.beginPath();ctx.moveTo(i,0);ctx.lineTo(i+30,CANVAS_SIZE);ctx.stroke();}
  // Grid
  ctx.strokeStyle='rgba(52,22,0,.62)'; ctx.lineWidth=.72;
  for(let i=0;i<SIZE;i++){
    const x=PAD+i*CELL,y=PAD+i*CELL;
    ctx.beginPath();ctx.moveTo(x,PAD);ctx.lineTo(x,PAD+(SIZE-1)*CELL);ctx.stroke();
    ctx.beginPath();ctx.moveTo(PAD,y);ctx.lineTo(PAD+(SIZE-1)*CELL,y);ctx.stroke();
  }
  ctx.strokeStyle='rgba(42,18,0,.82)'; ctx.lineWidth=1.8;
  ctx.strokeRect(PAD,PAD,(SIZE-1)*CELL,(SIZE-1)*CELL);
  // Stars
  ctx.fillStyle='rgba(48,20,0,.68)';
  [[3,3],[3,11],[7,7],[11,3],[11,11]].forEach(([r,c])=>{ctx.beginPath();ctx.arc(PAD+c*CELL,PAD+r*CELL,3.2,0,Math.PI*2);ctx.fill();});
  // Coords
  ctx.font='9.5px serif'; ctx.fillStyle='rgba(52,22,0,.42)';
  ctx.textAlign='center'; ctx.textBaseline='middle';
  const L='ABCDEFGHJKLMNOP';
  for(let i=0;i<SIZE;i++){
    ctx.fillText(L[i],PAD+i*CELL,PAD/2); ctx.fillText(L[i],PAD+i*CELL,CANVAS_SIZE-PAD/2);
    ctx.fillText(SIZE-i,PAD/2-1,PAD+i*CELL); ctx.fillText(SIZE-i,CANVAS_SIZE-PAD/2+1,PAD+i*CELL);
  }
  // Win line
  if(winLine.length>=2){
    const[r0,c0]=winLine[0],[r1,c1]=winLine[winLine.length-1];
    ctx.save(); ctx.strokeStyle='rgba(255,45,45,.82)'; ctx.lineWidth=4.5; ctx.lineCap='round';
    ctx.shadowColor='rgba(255,80,80,.45)'; ctx.shadowBlur=8;
    ctx.beginPath(); ctx.moveTo(PAD+c0*CELL,PAD+r0*CELL); ctx.lineTo(PAD+c1*CELL,PAD+r1*CELL); ctx.stroke();
    ctx.restore();
  }
  for(let r=0;r<SIZE;r++) for(let c=0;c<SIZE;c++)
    if(board[r][c]) drawPiece(r,c,board[r][c],lastMove&&lastMove[0]===r&&lastMove[1]===c);
}

function drawPiece(r,c,color,isLast) {
  const x=PAD+c*CELL, y=PAD+r*CELL, rad=CELL*.435;
  ctx.save();
  ctx.shadowColor='rgba(0,0,0,.52)'; ctx.shadowBlur=7; ctx.shadowOffsetX=2; ctx.shadowOffsetY=2.5;
  if(color===1){
    const g=ctx.createRadialGradient(x-rad*.32,y-rad*.32,rad*.04,x+rad*.1,y+rad*.1,rad);
    g.addColorStop(0,'#5a5a5a'); g.addColorStop(.22,'#1e1e1e'); g.addColorStop(1,'#040404');
    ctx.fillStyle=g;
  } else {
    const g=ctx.createRadialGradient(x-rad*.3,y-rad*.3,rad*.04,x+rad*.08,y+rad*.08,rad);
    g.addColorStop(0,'#ffffff'); g.addColorStop(.5,'#eee6d6'); g.addColorStop(1,'#b8a898');
    ctx.fillStyle=g;
  }
  ctx.beginPath(); ctx.arc(x,y,rad,0,Math.PI*2); ctx.fill(); ctx.restore();
  // Specular rim
  ctx.save(); ctx.strokeStyle=color===1?'rgba(255,255,255,.055)':'rgba(255,255,255,.52)'; ctx.lineWidth=.9;
  ctx.beginPath(); ctx.arc(x-rad*.18,y-rad*.18,rad*.62,Math.PI*1.1,Math.PI*1.65); ctx.stroke(); ctx.restore();
  // Last-move dot
  if(isLast){ctx.save();ctx.fillStyle=color===1?'rgba(255,88,88,.85)':'rgba(208,55,55,.75)';ctx.beginPath();ctx.arc(x,y,rad*.22,0,Math.PI*2);ctx.fill();ctx.restore();}
}

const DIRS=[[0,1],[1,0],[1,1],[1,-1]];
function checkWin(r,c,color){
  for(const[dr,dc]of DIRS){
    const line=[[r,c]];
    for(let d=1;d<5;d++){const nr=r+dr*d,nc=c+dc*d;if(nr<0||nr>=SIZE||nc<0||nc>=SIZE||board[nr][nc]!==color)break;line.push([nr,nc]);}
    for(let d=1;d<5;d++){const nr=r-dr*d,nc=c-dc*d;if(nr<0||nr>=SIZE||nc<0||nc>=SIZE||board[nr][nc]!==color)break;line.unshift([nr,nc]);}
    if(line.length>=5){winLine=line.slice(0,5);return true;}
  }
  return false;
}
function checkWinFast(r,c,color){
  for(const[dr,dc]of DIRS){let cnt=1;
    for(let d=1;d<5;d++){const nr=r+dr*d,nc=c+dc*d;if(nr<0||nr>=SIZE||nc<0||nc>=SIZE||board[nr][nc]!==color)break;cnt++;}
    for(let d=1;d<5;d++){const nr=r-dr*d,nc=c-dc*d;if(nr<0||nr>=SIZE||nc<0||nc>=SIZE||board[nr][nc]!==color)break;cnt++;}
    if(cnt>=5)return true;}return false;
}
function isBoardFull(){for(let r=0;r<SIZE;r++)for(let c=0;c<SIZE;c++)if(!board[r][c])return false;return true;}

const SC={FIVE:1000000,OPEN4:100000,CLOSE4:10000,OPEN3:5000,CLOSE3:500,OPEN2:200,CLOSE2:50,ONE:10};
function evaluateLine(cells){
  let s=0;
  for(const color of[1,2]){const opp=3-color,mul=color===aiColor?1:-1;let i=0;
    while(i<cells.length){if(cells[i]===opp){i++;continue;}let cnt=0,oL=(i===0||cells[i-1]===0)?1:0;
      while(i<cells.length&&cells[i]===color){cnt++;i++;}const oR=(i<cells.length&&cells[i]===0)?1:0;
      if(!cnt){i++;continue;}
      if(cnt>=5)s+=mul*SC.FIVE;else if(cnt===4)s+=mul*(oL+oR===2?SC.OPEN4:oL+oR===1?SC.CLOSE4:0);
      else if(cnt===3)s+=mul*(oL+oR===2?SC.OPEN3:oL+oR===1?SC.CLOSE3:0);
      else if(cnt===2)s+=mul*(oL+oR===2?SC.OPEN2:oL+oR===1?SC.CLOSE2:0);else s+=mul*SC.ONE;}}
  return s;
}
function evaluateBoard(){
  let t=0;
  for(let r=0;r<SIZE;r++)t+=evaluateLine(board[r].slice());
  for(let c=0;c<SIZE;c++){const col=[];for(let r=0;r<SIZE;r++)col.push(board[r][c]);t+=evaluateLine(col);}
  for(let s=0;s<2*SIZE-1;s++){const d=[];for(let r=0;r<SIZE;r++){const c=s-r;if(c>=0&&c<SIZE)d.push(board[r][c]);}t+=evaluateLine(d);}
  for(let s=0;s<2*SIZE-1;s++){const d=[];for(let r=0;r<SIZE;r++){const c=r+s-(SIZE-1);if(c>=0&&c<SIZE)d.push(board[r][c]);}t+=evaluateLine(d);}
  return t;
}
function getCandidates(range=2){
  const vis=Array.from({length:SIZE},()=>new Array(SIZE).fill(false)),moves=[];let has=false;
  for(let r=0;r<SIZE;r++)for(let c=0;c<SIZE;c++){if(!board[r][c])continue;has=true;
    for(let dr=-range;dr<=range;dr++)for(let dc=-range;dc<=range;dc++){const nr=r+dr,nc=c+dc;
      if(nr>=0&&nr<SIZE&&nc>=0&&nc<SIZE&&!board[nr][nc]&&!vis[nr][nc]){vis[nr][nc]=true;moves.push([nr,nc]);}}}
  return has?moves:[[7,7]];
}
function quickScore(r,c,color){
  let s=0;
  for(const[dr,dc]of DIRS){let cnt=1,open=0;
    for(let d=1;d<5;d++){const nr=r+dr*d,nc=c+dc*d;if(nr<0||nr>=SIZE||nc<0||nc>=SIZE)break;if(board[nr][nc]===color)cnt++;else{if(!board[nr][nc])open++;break;}}
    for(let d=1;d<5;d++){const nr=r-dr*d,nc=c-dc*d;if(nr<0||nr>=SIZE||nc<0||nc>=SIZE)break;if(board[nr][nc]===color)cnt++;else{if(!board[nr][nc])open++;break;}}
    if(cnt>=5)s+=SC.FIVE;else if(cnt===4)s+=(open>=1?SC.OPEN4:SC.CLOSE4);
    else if(cnt===3)s+=(open>=2?SC.OPEN3:SC.CLOSE3);else if(cnt===2)s+=(open>=1?SC.OPEN2:0);}
  return s;
}
function minimax(depth,alpha,beta,isMax,lr,lc){
  if(lr!==-1&&checkWinFast(lr,lc,isMax?playerColor:aiColor))return isMax?-SC.FIVE*(depth+1):SC.FIVE*(depth+1);
  if(depth===0)return evaluateBoard();
  const cands=getCandidates(2);if(!cands.length)return 0;
  const color=isMax?aiColor:playerColor;
  cands.sort((a,b)=>(quickScore(b[0],b[1],color)+quickScore(b[0],b[1],3-color))-(quickScore(a[0],a[1],color)+quickScore(a[0],a[1],3-color)));
  const top=Math.min(cands.length,depth>=3?8:12);
  if(isMax){let best=-Infinity;for(let i=0;i<top;i++){const[r,c]=cands[i];board[r][c]=aiColor;const v=minimax(depth-1,alpha,beta,false,r,c);board[r][c]=0;if(v>best)best=v;alpha=Math.max(alpha,best);if(beta<=alpha)break;}return best;}
  else{let best=Infinity;for(let i=0;i<top;i++){const[r,c]=cands[i];board[r][c]=playerColor;const v=minimax(depth-1,alpha,beta,true,r,c);board[r][c]=0;if(v<best)best=v;beta=Math.min(beta,best);if(beta<=alpha)break;}return best;}
}

function aiMove(){
  if(gameOver)return;
  canvas.classList.add('ai-thinking'); setStatusThinking();
  setTimeout(()=>{
    let best=null;
    if(difficulty===1){
      const cands=getCandidates(1);if(!cands.length){canvas.classList.remove('ai-thinking');return;}
      let bs=-Infinity;for(const[r,c]of cands){let s=quickScore(r,c,aiColor)*2+quickScore(r,c,playerColor)+Math.random()*3000;if(s>bs){bs=s;best=[r,c];}}
    } else {
      const depth=difficulty===2?2:difficulty===3?4:6;
      const cands=getCandidates(2);
      cands.sort((a,b)=>(quickScore(b[0],b[1],aiColor)+quickScore(b[0],b[1],playerColor))-(quickScore(a[0],a[1],aiColor)+quickScore(a[0],a[1],playerColor)));
      for(const[r,c]of cands){board[r][c]=aiColor;if(checkWinFast(r,c,aiColor)){board[r][c]=0;best=[r,c];break;}board[r][c]=0;}
      if(!best){for(const[r,c]of cands){board[r][c]=playerColor;if(checkWinFast(r,c,playerColor)){board[r][c]=0;best=[r,c];break;}board[r][c]=0;}}
      if(!best){let bv=-Infinity;const topN=Math.min(cands.length,depth>=5?10:15);for(let i=0;i<topN;i++){const[r,c]=cands[i];board[r][c]=aiColor;const v=minimax(depth-1,-Infinity,Infinity,false,r,c);board[r][c]=0;if(v>bv){bv=v;best=[r,c];}}}
    }
    canvas.classList.remove('ai-thinking');
    if(best)placeStone(best[0],best[1]);
  }, difficulty>=4?50:20);
}

function placeStone(r,c){
  if(gameOver||board[r][c])return;
  board[r][c]=currentTurn; lastMove=[r,c];
  moveHistory.push({r,c,color:currentTurn});
  sfxStone(currentTurn===1);
  if(checkWin(r,c,currentTurn)){
    drawBoard();gameOver=true;updateUndoBtn();
    if(currentTurn===playerColor){scores.win++;setStatusWin('你赢了！');sfxWin();}
    else{scores.lose++;setStatusWin('AI赢了');sfxLose();}
    updateScores();document.getElementById('boardFrame').classList.add('win-flash');return;
  }
  if(isBoardFull()){drawBoard();gameOver=true;scores.draw++;updateUndoBtn();updateScores();setStatusDraw();sfxDraw();return;}
  currentTurn=3-currentTurn;updateUndoBtn();drawBoard();updateStatus();
  if(currentTurn===aiColor)setTimeout(aiMove,100);
}
